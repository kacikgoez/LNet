    /**

    function escapeValue(depObj, name, value) {
        if (depObj.AssetMan && depObj.AssetMan[name]) {
            value = depObj.AssetMan[name];

    function doClone(node) {
        var clone = '';
        var ClickTaleDep = node.ClickTaleDep || {};
        if (ClickTaleDep.remove) {
            return '';
        } else if (ClickTaleDep.replace) {
            return '<!--' + node.outerHTML.replace(/<!--/g, '&lt;!--').replace(/-->/g, '--&gt;') + '-->';
        }

        var nodeName = node.nodeName.toLowerCase();
        if (nodeName === 'script') {
            return '<script><\/script>';
        }
        clone += ('<' + nodeName);
        var attributes = node.attributes;
        for (var i = 0, length = attributes.length; i < length; i++) {
            var attr = attributes[i];
            clone += (' ' + escapeValue(ClickTaleDep, attr.nodeName, attr.nodeValue));
        }
        if (window.ClickTaleSettings.PTC.emptyEls[nodeName]) {
            clone += ' \/>';
        } else {
            clone += '>';
            var child = node.firstChild;
            while (child) {
                var nodeType = child.nodeType;
                switch (nodeType) {
                    case 3:
                        var nodeValue = child.nodeValue
                        clone += ClickTaleDep.Text ? replaceStr(nodeValue) : nodeValue;
                        break;
                    case 8:
                        clone += '<!--' + child.nodeValue + '-->';
                        break;
                    default:
                        clone += doClone(child);
                }

                child = child.nextSibling;
            }
            clone += '<\/' + nodeName + '>';
        }


        return clone;

    }

    if (typeof times == "undefined") times = 10;
    if (--times < 0) {
        if (typeof failHandler === 'function') {
            failHandler();
        }
        return;
    }
    if (toCheckHandler()) {
        toDoHandler();
        return;
    }
    setTimeout(function () {
        window.ClickTaleSettings.PTC.doOnlyWhen(toDoHandler, toCheckHandler, interval, times, failHandler);
    }, interval);
